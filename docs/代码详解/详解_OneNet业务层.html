1. 模块概述

onenet.c / onenet.h 是智能灌溉系统的云平台业务逻辑层，位于 MqttKit 协议栈之上。该模块充当设备与 OneNet 云平台之间的桥梁，负责：

  • 设备登录 — 使用 PROID / TOKEN / DEVID 向 OneNet 进行 MQTT 连接鉴权
  • 属性上报 — 将土壤湿度、阈值、报警状态等属性定期发布到 OneNet 云
  • 命令接收与处理 — 订阅下行主题，接收并解析平台下发的阈值修改等命令
  • 心跳保活 — 定期发送 PINGREQ 维持 MQTT 连接
架构位置： main.c → onenet.c → MqttKit.c → esp8266.c → UART 硬件
常量说明
PROIDOneNet 产品 ID（作为 MQTT Username）
TOKEN设备密钥（作为 MQTT Password）
DEVID设备 ID（作为 MQTT Client ID）

2. 设备登录 — OneNet_DevLink

设备登录是上云的第一步。函数流程：构造 CONNECT 报文 → 通过 ESP8266 发送 → 等待 CONNACK 响应 → 检查返回码。

uint8_t OneNet_DevLink(void)
{
  MQTT_PACKET_STRUCTURE mqttPkt = MQTT_NewBuffer(512);
  uint8_t retry = 3;
  uint8_t result = 1;

  while (retry--) {
    // 使用 PROID/TOKEN/DEVID 构造 CONNECT 报文
    MQTT_PacketConnect(&mqttPkt,
      DEVID,       // Client ID
      PROID,       // Username
      TOKEN,       // Password
      60,          // Keep Alive = 60 秒
      1);          // Clean Session

    // 通过 ESP8266 发送 CONNECT
    ESP8266_SendData(mqttPkt._data, mqttPkt._len);

    // 等待 CONNACK 响应
    uint8_t *recvBuf = ESP8266_GetIPD(1000);
    if (recvBuf) {
      uint8_t type = MQTT_UnPacketRecv(recvBuf, 0);
      if (type == MQTT_CONNACK) {
        uint8_t ret = MQTT_UnPacketConnectAck(recvBuf);
        if (ret == 0) {
          result = 0;  // 登录成功
          break;
        }
      }
    }
  }

  MQTT_DeleteBuffer(&mqttPkt);
  return result;  // 0 = 成功，非 0 = 失败
}
容错机制：函数内部包含 3 次重试逻辑。主调函数（main.c）会持续重试直到登录成功，并在 OLED 上显示 "OneNet 连接中..." / "OneNet 连接成功"。
CONNACK 返回码速查：0 = 接受 ✓   1 = 协议版本拒绝   2 = 标识符被拒   3 = 服务器不可用   4 = 用户名/密码错误   5 = 未授权

3. 订阅下行主题 — OneNet_Subscribe

设备登录成功后，订阅 OneNet 平台的下行命令主题，用于接收属性设置指令：

uint8_t OneNet_Subscribe(void)
{
  MQTT_PACKET_STRUCTURE mqttPkt = MQTT_NewBuffer(256);

  // 构造订阅主题：$sys/{proid}/{devname}/thing/property/set
  char topic[128];
  sprintf(topic, "$sys/%s/%s/thing/property/set", PROID, DEVID);

  MQTT_PacketSubscribe(&mqttPkt, topic, MQTT_QOS_1, 1);
  ESP8266_SendData(mqttPkt._data, mqttPkt._len);

  // 等待 SUBACK
  uint8_t *recvBuf = ESP8266_GetIPD(2000);
  uint8_t result = 1;
  if (recvBuf) {
    uint8_t type = MQTT_UnPacketRecv(recvBuf, 0);
    if (type == MQTT_SUBACK) result = 0;
  }

  MQTT_DeleteBuffer(&mqttPkt);
  return result;
}
主题格式：OneNet 标准属性设置主题为 $sys/{proid}/{devname}/thing/property/set，其中 {proid} 和 {devname} 需替换为实际产品 ID 和设备名。

4. 发布属性数据 — OneNet_Publish

设备定期将传感器数据上报到 OneNet 平台。使用 PUBLISH 报文发送 JSON 格式的属性数据：

void OneNet_Publish(char *data)
{
  MQTT_PACKET_STRUCTURE mqttPkt = MQTT_NewBuffer(512);

  // 构造发布主题：$sys/{proid}/{devname}/thing/property/post
  char topic[128];
  sprintf(topic, "$sys/%s/%s/thing/property/post", PROID, DEVID);

  MQTT_PacketPublish(&mqttPkt, topic,
    (uint8_t *)data, strlen(data),
    MQTT_QOS_1, 1);

  ESP8266_SendData(mqttPkt._data, mqttPkt._len);
  MQTT_DeleteBuffer(&mqttPkt);
}

上报的 JSON 数据格式示例：

{
  "id": "123",
  "params": {
    "Humidity":   { "value": 68.5 },
    "Threshold":  { "value": 50 },
    "alarm":      { "value": false }
  }
}
属性名类型说明
Humidityfloat当前土壤湿度百分比（0.0 ~ 100.0）
Thresholdint湿度阈值（用户可远程修改）
alarmbool缺水报警标志

5. 心跳保活 — OneNet_SendPing

定期发送 PINGREQ 以保持 MQTT 连接不被服务器关闭：

void OneNet_SendPing(void)
{
  MQTT_PACKET_STRUCTURE mqttPkt = MQTT_NewBuffer(16);
  MQTT_PacketPing(&mqttPkt);
  ESP8266_SendData(mqttPkt._data, mqttPkt._len);
  MQTT_DeleteBuffer(&mqttPkt);
}
调度方式：在 main.c 主循环中通过 PingCount 计数控制。每 3000 个循环周期（约 30 秒）调用一次 OneNet_SendPing()。

6. 下行数据解析 — OneNet_RevPro（核心函数）

这是整个程序中最复杂的函数。每次主循环收到来自 ESP8266 的数据后，调用此函数进行完整的协议解析流程：识别报文类型 → 解析 PUBLISH 内容 → 提取 JSON 参数 → 更新全局变量 → 发送应答。

void OneNet_RevPro(uint8_t *recvBuf)
{
  uint8_t type = MQTT_UnPacketRecv(recvBuf, 0);
  char *topic = NULL;
  uint8_t *payload = NULL;
  uint16_t payloadLen = 0;
  uint8_t qos = 0;
  uint16_t packetId = 0;

  switch (type) {
    case MQTT_PUBLISH:
    case MQTT_CMD_TYPE: {
      // 解析 PUBLISH 报文
      MQTT_UnPacketPublish(recvBuf, &topic, &payload,
        &payloadLen, &qos, &packetId);

      // QoS 1 时需要回复 PUBACK
      if (qos == MQTT_QOS_1) {
        MQTT_PACKET_STRUCTURE ackPkt = MQTT_NewBuffer(16);
        MQTT_PacketPublishAck(&ackPkt, packetId);
        ESP8266_SendData(ackPkt._data, ackPkt._len);
        MQTT_DeleteBuffer(&ackPkt);
      }

      // 使用 cJSON 解析 JSON 负载
      cJSON *json = cJSON_Parse((const char *)payload);
      if (json) {
        cJSON *params = cJSON_GetObjectItem(json, "params");
        if (params) {
          cJSON *thresh = cJSON_GetObjectItem(params, "Threshold");
          if (thresh) {
            // 兼容两种 JSON 格式：直接数字 or {"value": 数字}
            if (thresh->type == cJSON_Number) {
              Humidity_Threshold = thresh->valuedouble;
            } else {
              cJSON *val = cJSON_GetObjectItem(thresh, "value");
              if (val && val->type == cJSON_Number)
                Humidity_Threshold = val->valuedouble;
            }
            // 持久化到 Flash
            Threshold_Save(Humidity_Threshold);
          }
        }
        cJSON_Delete(json);

        // 发送属性设置应答 set_reply
        char reply[128];
        sprintf(reply,
          "{\"id\":\"%s\",\"code\":200,\"msg\":\"success\"}",
          "");
        MQTT_SendPublish(
          "$sys/%s/%s/thing/property/set_reply",
          reply);
      }
      break;
    }

    case MQTT_PINGRESP:
      // 收到心跳响应，无需处理
      break;

    default:
      break;
  }
}
重要：函数内部分配的 MQTT 缓冲区（ackPkt）和 cJSON 对象（json）必须在使用后释放。每次主循环调用 OneNet_RevPro 后，上层还需释放 recvBuf（由 ESP8266_GetIPD 分配）。

7. 私有发布函数 — MQTT_SendPublish

这是一个独立于 MqttKit 的辅助函数，手动构造 PUBLISH 报文（逐字节写入），用于发送属性设置回复（set_reply）：

void MQTT_SendPublish(char *topic, char *payload)
{
  uint8_t buf[256];
  uint16_t topicLen = strlen(topic);
  uint16_t payloadLen = strlen(payload);
  uint16_t idx = 0;

  // 固定报头：PUBLISH 类型, QoS 1
  buf[idx++] = (MQTT_PUBLISH << 4) | (MQTT_QOS_1 << 1);

  // 剩余长度 = topic 长度 + payload 长度 + 2 (topicLen 字段)
  uint8_t remainLen = topicLen + payloadLen + 2;
  uint8_t lenBuf[4];
  uint8_t lenBytes = MQTT_DumpLength(remainLen, lenBuf);
  memcpy(buf + idx, lenBuf, lenBytes);
  idx += lenBytes;

  // Topic Name (UTF-8)
  buf[idx++] = (topicLen >> 8) & 0xFF;
  buf[idx++] = topicLen & 0xFF;
  memcpy(buf + idx, topic, topicLen);
  idx += topicLen;

  // Payload
  memcpy(buf + idx, payload, payloadLen);
  idx += payloadLen;

  ESP8266_SendData(buf, idx);
}
为什么不使用 MQTT_PacketPublish？此函数出现在日志调试阶段，为了简化依赖而直接手动构造报文，避免了创建和删除缓冲区的开销。实际效果与 MqttKit 的标准 API 一致。

8. cJSON 解析流程

下行命令的 JSON 解析过程涉及 cJSON 库的以下步骤：

  1. cJSON_Parse — 将原始字符串解析为 cJSON 对象树
  2. cJSON_GetObjectItem — 逐层提取 "params" → "Threshold" 节点
  3. 类型兼容 —
    • 若 Threshold 是 cJSON_Number（直接数值），直接读取 valuedouble
    • 若 Threshold 是 cJSON_Object（如 {"value": 50}），继续解析 value 子字段
  4. cJSON_Delete — 释放整个对象树，防止内存泄漏

JSON 解析的两种格式兼容示例：

// 格式一：直接数值（简化版）
{ "Threshold": 50 }

// 格式二：对象形式（标准 OneNet 属性格式）
{ "Threshold": { "value": 50 } }
内存注意：cJSON 解析后的对象树通过 cJSON_Parse 动态分配内存。每次解析完成后必须调用 cJSON_Delete 释放，否则会造成内存泄漏。在资源受限的 STM32 上，一次泄漏就可能导致系统崩溃。

onenet.c

#include "onenet.h"
#include "esp8266.h"
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include "cJSON.h"

uint16_t Humidity_Threshold = 50;  // 全局阈值，可在 main.c 中访问

/* ============== 设备登录 ============== */
uint8_t OneNet_DevLink(void)
{
  MQTT_PACKET_STRUCTURE mqttPkt = MQTT_NewBuffer(512);
  uint8_t retry = 3;
  uint8_t result = 1;

  while (retry--) {
    MQTT_PacketConnect(&mqttPkt, DEVID, PROID, TOKEN, 60, 1);
    ESP8266_SendData(mqttPkt._data, mqttPkt._len);

    uint8_t *recvBuf = ESP8266_GetIPD(1000);
    if (recvBuf) {
      if (MQTT_UnPacketRecv(recvBuf, 0) == MQTT_CONNACK) {
        if (MQTT_UnPacketConnectAck(recvBuf) == 0) {
          result = 0;
          break;
        }
      }
    }
  }

  MQTT_DeleteBuffer(&mqttPkt);
  return result;
}

/* ============== 订阅下行主题 ============== */
uint8_t OneNet_Subscribe(void)
{
  MQTT_PACKET_STRUCTURE mqttPkt = MQTT_NewBuffer(256);
  char topic[128];
  sprintf(topic, "$sys/%s/%s/thing/property/set", PROID, DEVID);
  MQTT_PacketSubscribe(&mqttPkt, topic, MQTT_QOS_1, 1);
  ESP8266_SendData(mqttPkt._data, mqttPkt._len);

  uint8_t result = 1;
  uint8_t *recvBuf = ESP8266_GetIPD(2000);
  if (recvBuf && MQTT_UnPacketRecv(recvBuf, 0) == MQTT_SUBACK)
    result = 0;

  MQTT_DeleteBuffer(&mqttPkt);
  return result;
}

/* ============== 私有发布函数 ============== */
void MQTT_SendPublish(char *topic, char *payload)
{
  uint8_t buf[256];
  uint16_t topicLen = strlen(topic);
  uint16_t payloadLen = strlen(payload);
  uint16_t idx = 0;

  buf[idx++] = (MQTT_PUBLISH << 4) | (MQTT_QOS_1 << 1);
  uint8_t remain = topicLen + payloadLen + 2;
  uint8_t lb[4], lc = MQTT_DumpLength(remain, lb);
  memcpy(buf + idx, lb, lc); idx += lc;
  buf[idx++] = (topicLen >> 8) & 0xFF;
  buf[idx++] = topicLen & 0xFF;
  memcpy(buf + idx, topic, topicLen); idx += topicLen;
  memcpy(buf + idx, payload, payloadLen); idx += payloadLen;
  ESP8266_SendData(buf, idx);
}

/* ============== 下行数据解析（核心） ============== */
void OneNet_RevPro(uint8_t *recvBuf)
{
  uint8_t type = MQTT_UnPacketRecv(recvBuf, 0);
  char *topic = NULL;
  uint8_t *payload = NULL;
  uint16_t payloadLen = 0;
  uint8_t qos = 0;
  uint16_t packetId = 0;

  switch (type) {
    case MQTT_PUBLISH:
    case MQTT_CMD_TYPE: {
      MQTT_UnPacketPublish(recvBuf, &topic, &payload,
        &payloadLen, &qos, &packetId);

      if (qos == MQTT_QOS_1) {
        MQTT_PACKET_STRUCTURE ack = MQTT_NewBuffer(16);
        MQTT_PacketPublishAck(&ack, packetId);
        ESP8266_SendData(ack._data, ack._len);
        MQTT_DeleteBuffer(&ack);
      }

      cJSON *json = cJSON_Parse((const char *)payload);
      if (json) {
        cJSON *params = cJSON_GetObjectItem(json, "params");
        if (params) {
          cJSON *t = cJSON_GetObjectItem(params, "Threshold");
          if (t) {
            if (t->type == cJSON_Number)
              Humidity_Threshold = t->valuedouble;
            else {
              cJSON *v = cJSON_GetObjectItem(t, "value");
              if (v && v->type == cJSON_Number)
                Humidity_Threshold = v->valuedouble;
            }
            Threshold_Save(Humidity_Threshold);
          }
        }
        cJSON_Delete(json);

        char reply[128];
        sprintf(reply,
          "{\"id\":\"%s\",\"code\":200,\"msg\":\"success\"}",
          "");
        MQTT_SendPublish(
          "$sys/%s/%s/thing/property/set_reply",
          reply);
      }
      break;
    }
    default: break;
  }
